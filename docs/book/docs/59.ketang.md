   let startY:number;//&#x6309;&#x4E0B;&#x65F6;&#x5019;&#x7684;&#x521D;&#x59CB;&#x7EB5;&#x5750;&#x6807;

       if(element.offsetTop == originalTop &amp;&amp;element.scrollTop == 0){

     let pageY  = event.touches[0].pageY;
